335  assert_param(IS_MPU_ACCESS_BUFFERABLE(MPU_Init->IsBufferable));
336  assert_param(IS_MPU_SUB_REGION_DISABLE(MPU_Init->SubRegionDisable));
337  assert_param(IS_MPU_REGION_SIZE(MPU_Init->Size));
338 
339  MPU->RBAR = MPU_Init->BaseAddress;
340  MPU->RASR = ((uint32_t)MPU_Init->DisableExec << MPU_RASR_XN_Pos) |
341  ((uint32_t)MPU_Init->AccessPermission << MPU_RASR_AP_Pos) |
342  ((uint32_t)MPU_Init->TypeExtField << MPU_RASR_TEX_Pos) |
343  ((uint32_t)MPU_Init->IsShareable << MPU_RASR_S_Pos) |

347  ((uint32_t)MPU_Init->Size << MPU_RASR_SIZE_Pos) |
